                let alerts = [];
                tbody.innerHTML = res[0].values.map(v => {
                    const failRate = ((v[2]/v[1])*100).toFixed(1);
                    const limit = State.ropConfig.leadTime; //허용 임계 불량률
                    const isDanger = failRate >= limit;
                    const price = State.cachedMasterMap[v[0]]?.price || 0;
                    if(isDanger) alerts.push(v[0]);
                    return `<tr class="${isDanger?'bg-rose-950/20 text-rose-400':''}">
                        <td class="p-4 font-bold">${v[0]}</td><td class="p-4 text-right">${v[1].toLocaleString()}</td><td class="p-4 text-right text-red-400">${v[2].toLocaleString()}</td><td class="p-4 text-right text-emerald-400">${(v[1]-v[2]).toLocaleString()}</td>
